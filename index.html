    function findRegion() {

      const input = document.getElementById('cityInput').value.toLowerCase();

      const resultDiv = document.getElementById('result');

      let foundRegions = [];

      for (const region in cityRegionData) {

        if (cityRegionData[region].some(city => city.toLowerCase().includes(input))) {

          foundRegions.push(region);

        }

      }

      if (foundRegions.length > 0) {

        resultDiv.innerHTML = `Regions for '${input}': <br> ${foundRegions.join('<br>')}`;

      } else {

        resultDiv.innerHTML = "City not found in the database.";

      }

    }
